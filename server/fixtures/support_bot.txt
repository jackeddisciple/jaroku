"""Support agent: looks up orders in Postgres and drafts a reply."""
from __future__ import annotations

from langchain_core.messages import HumanMessage, SystemMessage, ToolMessage
from langgraph.graph import END, START, MessagesState, StateGraph
from langgraph.prebuilt import ToolNode

from .prompts import SYSTEM_PROMPT
from .tools import TOOLS


class AgentState(MessagesState):
    orders: list[str]


def build_graph(llm):
    model = llm.bind_tools(TOOLS)

    def call_model(state: AgentState):
        messages = state["messages"]
        if not any(isinstance(m, SystemMessage) for m in messages):
            messages = [SystemMessage(SYSTEM_PROMPT), *messages]
        return {"messages": [model.invoke(messages)]}

    def record_orders(state: AgentState):
        last = state["messages"][-1]
        orders = list(state.get("orders") or [])
        if isinstance(last, ToolMessage) and last.name == "pg_query":
            orders.append(str(last.content)[:200])
        return {"orders": orders}

    def should_continue(state: AgentState) -> str:
        last = state["messages"][-1]
        return "tools" if getattr(last, "tool_calls", None) else END

    graph = StateGraph(AgentState)
    graph.add_node("agent", call_model)
    graph.add_node("tools", ToolNode(TOOLS))
    graph.add_node("record_orders", record_orders)
    graph.add_edge(START, "agent")
    graph.add_conditional_edges("agent", should_continue, {"tools": "tools", END: END})
    graph.add_edge("tools", "record_orders")
    graph.add_edge("record_orders", "agent")
    return graph.compile()


def build_initial_state(user_input: str) -> dict:
    return {"messages": [HumanMessage(user_input)], "orders": []}

from __future__ import annotations

from langchain_core.tools import tool


@tool
def summarize_issue(text: str, max_words: int = 40) -> str:
    """Condense a customer message to its core issue in at most `max_words` words."""
    words = text.split()
    if not words:
        return "The message is empty."
    if len(words) <= max_words:
        return text.strip()
    return " ".join(words[:max_words]) + "..."
